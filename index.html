  <script>

    function submitForm() {

      const yourName = document.getElementById('yourName').value;

      const crushName = document.getElementById('crushName').value;

      const resultText = document.getElementById('result');


      if (!yourName || !crushName) {

        alert("이름을 모두 입력해주세요!");

        return;

      }


      // 랜덤 확률 생성

      const percent = Math.floor(Math.random() * 51) + 50;

      resultText.innerText = `😍 ${yourName}와(과) ${crushName}의 짝사랑 성공 확률은 ${percent}%입니다!`;


      // 이메일 발송

      fetch("https://formspree.io/f/xanegwav", {

        method: "POST",

        headers: {

          "Accept": "application/json",

          "Content-Type": "application/json"

        },

        body: JSON.stringify({

          name: yourName,

          crush: crushName,

          message: `${yourName} -> ${crushName}`,

        })

      });

    }
